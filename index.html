Arsitektur Serverless Google Apps Script

Aplikasi PROPemperda ini dirancang agar dapat berjalan secara mandiri (Standalone HTML/JS) maupun diintegrasikan sepenuhnya dengan Google Workspace (Google Sheets sebagai Database & Google Drive sebagai penyimpanan file kajian PDF/DOCX) tanpa biaya langganan server.

Langkah Deploy ke Google Apps Script:
  1. Buka Google Drive dan buat Spreadsheet baru bernama "DB_PROPemperda_Pasaman".
  2. Pilih menu Ekstensi > Apps Script.
  3. Salin seluruh kode backend dari kotak di sebelah kanan ke dalam file Code.gs.
  4. Klik tombol Terapkan (Deploy) > Deployment baru.
  5. Pilih jenis deployment: Aplikasi Web (Web App).
  6. Atur Akses: "Siapa saja (Anyone)" agar aplikasi dapat diakses publik dengan otentikasi internal.
  7. Salin Web App URL dan tempelkan ke konfigurasi koneksi di bawah ini!
Uji Koneksi Live Google Apps Script
Saat ini aplikasi beroperasi dalam modus Hybrid LocalStorage + Google Sheets.
Sinkronisasi Data Awal (Demo Seed) Unggah seluruh data contoh di aplikasi ini langsung menjadi baris data awal di Google Sheets Anda.
Kode Backend (Code.gs) untuk Google Apps Script 
// =========================================================
// BACKEND GOOGLE APPS SCRIPT (Code.gs) - PROPEMPERDA PASAMAN
// =========================================================

const FOLDER_ID = "YOUR_GOOGLE_DRIVE_FOLDER_ID";
const SHEET_NAME = "Data_Usulan";

function doGet(e) {
  var action = e.parameter.action || "read";
  if (action === "read") {
    return ContentService.createTextOutput(JSON.stringify(getAllUsulan()))
      .setMimeType(ContentService.MimeType.JSON);
  }
  return HtmlService.createHtmlOutput("PROPemperda Pasaman API Running.");
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);
    
    // Simpan berkas ke Google Drive jika ada file base64
    var fileUrl = "";
    if (data.fileBase64) {
      var folder = DriveApp.getFolderById(FOLDER_ID);
      var blob = Utilities.newBlob(Utilities.base64Decode(data.fileBase64), data.fileMime, data.fileName);
      var file = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      fileUrl = file.getUrl();
    }
    
    var row = [
      new Date(), data.id, data.opd, data.pic, data.wa,
      data.jenis, data.statusRegulasi, data.judul, data.urgensi,
      data.ruangLingkup, data.isDelegasi, data.targetSelesai,
      "Draft", fileUrl
    ];
    sheet.appendRow(row);
    
    return ContentService.createTextOutput(JSON.stringify({status: "success", fileUrl: fileUrl}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function getAllUsulan() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  if (!sheet) return [];
  var data = sheet.getDataRange().getValues();
  var headers = data.shift();
  return data.map(function(row) {
    return {
      timestamp: row[0], id: row[1], opd: row[2], pic: row[3],
      wa: row[4], jenis: row[5], status: row[12], fileUrl: row[13]
    };
  });
}
